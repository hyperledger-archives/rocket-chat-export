      - $PWD/crypto-config/ordererOrganizations/example.com/orderers/orderer.example.com/msp:/var/hyperledger/orderer/msp

    ports:

    ports:

    command: sh -c 'fabric-ca-server start --ca.certfile /etc/hyperledger/fabric-ca-server-config/ca.org1.example.com-cert.pem --ca.keyfile /etc/hyperledger/fabric-ca-server-config/cf49dac80287b3cf78724ba08bd5efcfaff27ce2efb07e69275cf1074345ec8a_sk -b admin:adminpw -d'
