asset Commodity identified by tradingSymbol {
  o String tradingSymbol
  o String description
  o String cmExchange
  o Double quantity
  --> Trader owner
}

participant Trader identified by traderId {
  o String traderId
  o String firstName
  o String lastName
}

  --> Commodity metal
  --> Trader newOwner
}
